'use strict';

function bento(hour) {
  if (hour === 9 || hour === 15) {
    return "お弁当を注文すると1個おまけ";
  } else if (hour >= 19 && hour < 21) {
    return "お弁当30%オフ!";
  } else {
    return "お弁当はいかがですか?";
  }
}

count hour = new Date().getHours();
alert(bento(hour));
